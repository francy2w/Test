-- Función para obtener el mapa
function getMap()
    for _, obj in ipairs(workspace:GetChildren()) do
        if obj:FindFirstChild("CoinContainer") and obj:FindFirstChild("Spawns") then
            return obj
        end
    end
    return nil
end

-- Función para obtener el murderer
local function getMurderer()
    for _, player in ipairs(Players:GetPlayers()) do
        if player.Character and player.Backpack and 
           (player.Character:FindFirstChild("Knife") or player.Backpack:FindFirstChild("Knife")) then
            return player
        end
    end
    return nil
end

-- Función para obtener el sheriff
local function getSheriff()
    for _, player in ipairs(Players:GetPlayers()) do
        if player.Character and player.Backpack and 
           (player.Character:FindFirstChild("Gun") or player.Backpack:FindFirstChild("Gun")) then
            return player
        end
    end
    return nil
end

-- Kill Aura
local killAuraEnabled = false
local killAuraConnection = nil
local killAuraReach = 7
local killAuraAngle = 60

MainTab:Toggle({
    Name = "Kill Aura",
    Flag = "KillAura",
    Callback = function(state)
        killAuraEnabled = state
        
        if state then
            killAuraConnection = RunService.Heartbeat:Connect(function()
                local char = localPlayer.Character
                if not char then return end
                
                local hrp = char:FindFirstChild("HumanoidRootPart")
                local knife = char:FindFirstChild("Knife")
                
                if hrp and knife and knife:FindFirstChild("Stab") then
                    for _, player in ipairs(Players:GetPlayers()) do
                        if player ~= localPlayer and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                            local targetHrp = player.Character.HumanoidRootPart
                            local humanoid = player.Character:FindFirstChild("Humanoid")
                            
                            if targetHrp and humanoid and humanoid.Health > 0 then
                                local distance = (targetHrp.Position - hrp.Position).Magnitude
                                local dotProduct = (targetHrp.Position - hrp.Position).Unit:Dot(hrp.CFrame.LookVector)
                                local angle = math.acos(math.clamp(dotProduct, -1, 1))
                                local angleDeg = math.deg(angle)
                                
                                if distance < killAuraReach and angleDeg < killAuraAngle / 2 then
                                    targetHrp.Anchored = true
                                    targetHrp.CFrame = hrp.CFrame + hrp.CFrame.LookVector * 2
                                    task.wait(0.1)
                                    knife.Stab:FireServer("Slash")
                                    task.wait(0.1)
                                    targetHrp.Anchored = false
                                end
                            end
                        end
                    end
                end
            end)
        elseif killAuraConnection then
            killAuraConnection:Disconnect()
            killAuraConnection = nil
        end
    end
})

-- Loop principal de farm
RunService.Heartbeat:Connect(function()
    if not autoFarmEnabled then return end
    
    local currentTime = tick()
    if currentTime - lastFarmTime >= farmInterval then
        local char = localPlayer.Character
        if not char or not char:FindFirstChild("HumanoidRootPart") then return end
        
        local hrp = char.HumanoidRootPart
        local coin = nil
        
        -- Buscar coins
        local coinsFolder = workspace:FindFirstChild("CoinContainer") or getMap()
        if coinsFolder then
            coinsFolder = coinsFolder:FindFirstChild("CoinContainer")
        end
        
        if coinsFolder then
            if farmMode == "Nearest" then
                local minDistance = 50
                for _, coinPart in ipairs(coinsFolder:GetChildren()) do
                    if coinPart:IsA("BasePart") and coinPart ~= lastCoin then
                        local distance = (hrp.Position - coinPart.Position).Magnitude
                        if distance < minDistance then
                            coin = coinPart
                            minDistance = distance
                        end
                    end
                end
            elseif farmMode == "Random" then
                local availableCoins = {}
                for _, coinPart in ipairs(coinsFolder:GetChildren()) do
                    if coinPart:IsA("BasePart") and coinPart ~= lastCoin then
                        table.insert(availableCoins, coinPart)
                    end
                end
                if #availableCoins > 0 then
                    coin = availableCoins[math.random(1, #availableCoins)]
                end
            end
        end
        
        if coin then
            hrp.CFrame = coin.CFrame
            lastCoin = coin
            lastFarmTime = currentTime
        end
    end
end)

local espSettings = {
    Murder = false,
    Sheriff = false,
    Innocent = false,
    Gun = false
}

local espHighlights = {}

local function getPlayerRole(player)
    if player == getMurderer() then
        return "Murder"
    elseif player == getSheriff() then
        return "Sheriff"
    else
        return "Innocent"
    end
end

local function getRoleColor(role)
    if role == "Murder" then
        return Color3.fromRGB(255, 0, 0)
    elseif role == "Sheriff" then
        return Color3.fromRGB(0, 150, 255)
    else
        return Color3.fromRGB(0, 255, 0)
    end
end

local function updateESP()
    -- Limpiar ESP antiguo
    for _, highlight in pairs(espHighlights) do
        if highlight then
            highlight:Destroy()
        end
    end
    espHighlights = {}
    
    -- Aplicar nuevo ESP
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= localPlayer and player.Character then
            local role = getPlayerRole(player)
            
            if espSettings[role] then
                local highlight = Instance.new("Highlight")
                highlight.Name = "CHEIXHUB_ESP"
                highlight.Adornee = player.Character
                highlight.FillTransparency = 0.5
                highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                highlight.FillColor = getRoleColor(role)
                highlight.OutlineColor = getRoleColor(role)
                highlight.Parent = player.Character
                
                espHighlights[player.Name] = highlight
            end
        end
    end
end

ESPTab:Toggle({
    Name = "Murder ESP",
    Flag = "MurderESP",
    Callback = function(state)
        espSettings.Murder = state
        updateESP()
    end
})

ESPTab:Toggle({
    Name = "Sheriff ESP",
    Flag = "SheriffESP",
    Callback = function(state)
        espSettings.Sheriff = state
        updateESP()
    end
})

ESPTab:Toggle({
    Name = "Innocent ESP",
    Flag = "InnocentESP",
    Callback = function(state)
        espSettings.Innocent = state
        updateESP()
    end
})

ESPTab:Toggle({
    Name = "Gun ESP",
    Flag = "GunESP",
    Callback = function(state)
        espSettings.Gun = state
        
        if state then
            local highlight = Instance.new("Highlight")
            highlight.Name = "CHEIXHUB_GUN_ESP"
            highlight.OutlineTransparency = 1
            highlight.FillColor = Color3.fromRGB(0, 255, 0)
            highlight.FillTransparency = 0.3
            highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
            
            local function checkGun(obj)
                if obj.Name == "GunDrop" and obj:IsA("BasePart") then
                    highlight.Adornee = obj
                    highlight.Parent = CoreGui
                    highlight.Enabled = true
                end
            end
            
            for _, obj in ipairs(workspace:GetDescendants()) do
                checkGun(obj)
            end
            
            workspace.DescendantAdded:Connect(checkGun)
            
            espHighlights["Gun"] = highlight
        else
            if espHighlights["Gun"] then
                espHighlights["Gun"]:Destroy()
                espHighlights["Gun"] = nil
            end
        end
    end
})

-- Actualizar ESP cuando se añaden jugadores
Players.PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(function()
        updateESP()
    end)
end)

localPlayer.CharacterAdded:Connect(function()
    updateESP()
end)
